python -m venv rag_demo
rag_demo\Scripts\activate

# Upgrade pip
python -m pip install --upgrade pip

1. Vector Database Setup (Chroma)
# Install Chroma and dependencies
pip install chromadb==0.4.22
pip install sentence-transformers==2.2.2
pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cpu

2. Local LLM Setup (Ollama)
Step-by-step Ollama installation:

Download Ollama for Windows:

Go to https://ollama.ai/download/windows
Download and run the installer


Install a suitable model:
# Open Command Prompt or PowerShell
# Install Llama 3.1 8B (good balance of speed/quality for 16GB RAM)
ollama pull llama3.1:8b

# Alternative: Lighter model for faster inference
ollama pull phi3:mini

# Test the installation
ollama run llama3.1:8b "Hello, how are you?"
